Overview

Represents a quote of a message.

Quotes have an associated server, channel, and user. They also have a rep value which is shown in the quote.

Quotes are posted to a channel named #quotes if it exists in the server.

The bot's quote message ID is cached and will be updated as the users rep the message.

def before_destroy
  sleep 1 # seems to dodge some kind of race condition?
  quote_channel = message.server.channels
                         .find { |c| c.channel_name == 'quotes' }
                         .channel
  unless quote_channel.nil?
    quote_message = quote_channel.load_message(quote_message_id)
    quote_message.delete unless quote_message.nil?
  end
  Discordrb::LOGGER.info "deleted quote: #{inspect}"
end

def update
  quote_channel = message.server.channels
                         .find { |c| c.channel_name == 'quotes' }
                         .channel
  unless quote_channel.nil?
    quote = "**##{id}📌** #{message.message_content}"
    quote << " (attachment: #{message.attachment_url})" unless message.attachment_url.nil?
    quote << " - #{message.timestamp.utc} by #{message.user.user.mention}"
    quote << " in #{message.channel.channel.mention}"
    quote << " **👌x#{rep}**"
    quote_message = quote_channel.load_message(quote_message_id)
    if quote_message.nil?
      quote_message = quote_channel.send_message quote
      self.quote_message_id = quote_message.id
      save
    else
      quote_message.edit quote
      Discordrb::LOGGER.info "updated quote: #{inspect}"
    end
  end
end
