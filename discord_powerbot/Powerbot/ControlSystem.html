def report_status
  update
  hq_distance = eddb_system.distance power.eddb_system
  last_update = powerplay_logs.last
  fort_percentage = (progress_fortification.to_f / trigger_fortification.to_f) * 100.0
  report = []
  report << '```hs'
  report << "#{name} (#{hq_distance.round(2)} ly)"
  report << " progress: #{progress_fortification} T / #{trigger_fortification} T (#{fort_percentage.round(2)}%)"
  report << "    value: #{cc_value.round} CC"
  report << ''
  report << "last updated by: #{last_update.user.cmdr_name} (#{last_update.timestamp.utc})"
  report << "         method: #{last_update.action}"
  report << '```'
  report << "`backhaul url:` #{backhaul_url}" unless backhaul_url.nil?
  report.join("\n")
end

def update
  total = 0
  unless powerplay_logs.empty?
    set   = powerplay_logs.select { |g| g.action == 'setforts' }.last
    logs  = powerplay_logs.select { |g| g.timestamp > set.timestamp }
    total = set.merits + logs.inject(0) { |sum, t| sum + t.merits }
  end
  self.progress_fortification = total
  total
end
