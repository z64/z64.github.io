def cc_overhead
  #-1*MIN((11.5 * B2/42)^3,(5.4 * 11.5 * B2))
  systems = control_systems.length
  part_cubic = (11.5 * systems / 42)**3
  part_linear    = 11.5 * systems * 5.4
  overhead       = 0.0
  overhead = if part_cubic < part_linear
               part_cubic
             else
               part_linear
             end
  { total: overhead, system: overhead / systems }
end
